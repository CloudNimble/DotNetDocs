## LINQ Example

<DotNetRunner
  title="LINQ Basics"
  height="250px"
  initialCode={`using System;
using System.Linq;

var numbers = Enumerable.Range(1, 10);
var evens = numbers.Where(n => n % 2 == 0);
var squares = evens.Select(n => n * n);

Console.WriteLine("Even squares from 1-10:");
Console.WriteLine(string.Join(", ", squares));`}
/>
